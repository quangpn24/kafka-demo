# create jdbc source connector
curl -X POST -H "Content-Type: application/json" --data @source/jdbc/mysql-user-source.json http://localhost:8083/connectors

# create debezium source connector
curl -X POST -H "Content-Type: application/json" --data @source/debezium/mysql-user-source.json http://localhost:8083/connectors

# create es sink connector
curl -X POST -H "Content-Type: application/json" --data @sink/elasticsearch/es-todo-sink.json http://localhost:8083/connectors


# update config for jdbc source connector
curl -X PUT -H "Content-Type: application/json" --data @source/jdbc/config.json http://localhost:8083/connectors/mysql-todo-source-connector/config

# update config for elasticsearch sink connector
curl -X PUT -H "Content-Type: application/json" --data @sink/elasticsearch/config.json http://localhost:8083/connectors/es-todo-sink-connector/config

# update config for debezium source connector
curl -X PUT -H "Content-Type: application/json" --data @source/debezium/config.json http://localhost:8083/connectors/debezium-user-source-connector/config


# delete connectors
curl -X DELETE http://localhost:8083/connectors/mysql-todo-source-connector

# get status
curl -s -X GET http://localhost:8083/connectors/mysql-todo-source-connector/status

# kafka
// consume with key
kafka-console-consumer --bootstrap-server kafka:9092 --topic schemahistory.debezium_connector --property print.key=true --from-beginning
