# create jdbc source connector
curl -X POST -H "Content-Type: application/json" --data @elastic_sink/es-todo-sink.json http://localhost:8083/connectors

# create es sink connector
curl -X POST -H "Content-Type: application/json" --data @jdbc_source/mysql-todo-source.json http://localhost:8083/connectors

# update config for jdbc source connector
curl -X PUT -H "Content-Type: application/json" --data @jdbc_source/config.json http://localhost:8083/connectors/mysql-todo-source-connector/config

# update config for elasticsearch sink connector
curl -X PUT -H "Content-Type: application/json" --data @elastic_sink/config.json http://localhost:8083/connectors/es-todo-sink-connector/config

# delete connectors
curl -X DELETE http://localhost:8083/connectors/mysql-todo-source-connector

# get status
curl -s -X GET http://localhost:8083/connectors/mysql-todo-source-connector/status
